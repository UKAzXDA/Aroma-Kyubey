git init

git status - Verificar o estado atual do repositório.
git add . - Adicionar todas as mudanças ao staging area.
git commit -m "sua mensagem" - Commitar as mudanças com uma mensagem explicativa.
git push - Empurrar as mudanças para o repositório remoto.

git checkout main
git pull origin main
git tag -a v1.1.0 -m "Release version 1.1.0"
git push origin v1.1.0

git remote set-url origin git@github.com:UKAzXDA/Aroma-Kyubey.git
git push origin main --force
git push origin main
